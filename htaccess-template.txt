RewriteEngine On

# Block access to sensitive files
<Files ".env">
    Order Allow,Deny
    Deny from all
</Files>

# Main URL rewriting for IOC system
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^(.*)$ index.php?url=$1 [QSA,L]

# Set default charset
AddDefaultCharset UTF-8
